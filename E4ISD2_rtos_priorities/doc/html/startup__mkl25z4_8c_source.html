258 //*****************************************************************************
259 // Functions to carry out the initialization of RW and BSS data sections. These
260 // are written as separate functions rather than being inlined within the
261 // ResetISR() function in order to cope with MCUs with multiple banks of
262 // memory.
263 //*****************************************************************************
264 __attribute__ ((section(".after_vectors.init_data")))
265 void data_init(unsigned int romstart, unsigned int start, unsigned int len) {
266  unsigned int *pulDest = (unsigned int*) start;
267  unsigned int *pulSrc = (unsigned int*) romstart;
268  unsigned int loop;
269  for (loop = 0; loop < len; loop = loop + 4)
270  *pulDest++ = *pulSrc++;
271 }

273 __attribute__ ((section(".after_vectors.init_bss")))
274 void bss_init(unsigned int start, unsigned int len) {
275  unsigned int *pulDest = (unsigned int*) start;
276  unsigned int loop;

281 //*****************************************************************************
282 // The following symbols are constructs generated by the linker, indicating
283 // the location of various points in the "Global Section Table". This table is
284 // created by the linker via the Code Red managed linker script mechanism. It
285 // contains the load address, execution address and length of each RW data
286 // section and the execution and length of each BSS (zero initialized) section.
287 //*****************************************************************************
288 extern unsigned int __data_section_table;
289 extern unsigned int __data_section_table_end;
290 extern unsigned int __bss_section_table;
291 extern unsigned int __bss_section_table_end;

293 //*****************************************************************************
294 // Reset entry point for your code.
295 // Sets up a simple runtime environment and initializes the C/C++
296 // library.
297 //*****************************************************************************
298 __attribute__ ((section(".after_vectors.reset")))
299 void ResetISR(void) {
300 
301  // Disable interrupts
302  __asm volatile ("cpsid i");
303 
304 #if defined (__USE_CMSIS)
305 // If __USE_CMSIS defined, then call CMSIS SystemInit code
306  SystemInit();
307 #else
308  // Disable Watchdog
309  // SIM->COPC register: COPT=0,COPCLKS=0,COPW=0
310  *((volatile unsigned int *)0x40048100) = 0x00u;
311 #endif // (__USE_CMSIS)
312 
313  //
314  // Copy the data sections from flash to SRAM.
315  //
316  unsigned int LoadAddr, ExeAddr, SectionLen;
317  unsigned int *SectionTableAddr;
318 
319  // Load base address of Global Section Table
320  SectionTableAddr = &__data_section_table;
321 
322  // Copy the data sections from flash to SRAM.
323  while (SectionTableAddr < &__data_section_table_end) {
324  LoadAddr = *SectionTableAddr++;
325  ExeAddr = *SectionTableAddr++;
326  SectionLen = *SectionTableAddr++;
327  data_init(LoadAddr, ExeAddr, SectionLen);
328  }
329 
330  // At this point, SectionTableAddr = &__bss_section_table;
331  // Zero fill the bss segment
332  while (SectionTableAddr < &__bss_section_table_end) {
333  ExeAddr = *SectionTableAddr++;
334  SectionLen = *SectionTableAddr++;
335  bss_init(ExeAddr, SectionLen);
336  }
337 
338 #if !defined (__USE_CMSIS)
339 // Assume that if __USE_CMSIS defined, then CMSIS SystemInit code
340 // will setup the VTOR register
341 
342  // Check to see if we are running the code from a non-zero
343  // address (eg RAM, external flash), in which case we need
344  // to modify the VTOR register to tell the CPU that the
345  // vector table is located at a non-0x0 address.
346  unsigned int * pSCB_VTOR = (unsigned int *) 0xE000ED08;
347  if ((unsigned int *)g_pfnVectors!=(unsigned int *) 0x00000000) {
348  *pSCB_VTOR = (unsigned int)g_pfnVectors;
349  }
350 #endif // (__USE_CMSIS)
351 
352 #if defined (__cplusplus)
353  //
354  // Call C++ library initialisation
355  //
356  __libc_init_array();
357 #endif
358 
359  // Reenable interrupts
360  __asm volatile ("cpsie i");
361 
362 #if defined (__REDLIB__)
363  // Call the Redlib library, which in turn calls main()
364  __main();
365 #else
366  main();
367 #endif
368 
369  //
370  // main() shouldn't return, but if it does, we'll just enter an infinite loop
371  //
372  while (1) {
373  ;
374  }
375 }
