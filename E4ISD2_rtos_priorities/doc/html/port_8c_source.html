125 StackType_t *pxPortInitialiseStack( StackType_t *pxTopOfStack, TaskFunction_t pxCode, void *pvParameters )
126 {
127  /* Simulate the stack frame as it would be created by a context switch
128  interrupt. */
129  pxTopOfStack--; /* Offset added to account for the way the MCU uses the stack on entry/exit of interrupts. */
130  *pxTopOfStack = portINITIAL_XPSR; /* xPSR */
131  pxTopOfStack--;
132  *pxTopOfStack = ( StackType_t ) pxCode; /* PC */
133  pxTopOfStack--;
134  *pxTopOfStack = ( StackType_t ) portTASK_RETURN_ADDRESS; /* LR */
135  pxTopOfStack -= 5; /* R12, R3, R2 and R1. */
136  *pxTopOfStack = ( StackType_t ) pvParameters; /* R0 */
137  pxTopOfStack -= 8; /* R11..R4. */
138 
139  return pxTopOfStack;
140 }

143 static void prvTaskExitError( void )
144 {
145  /* A function that implements a task must not exit or attempt to return to

164 void vPortStartFirstTask( void )
165 {
166  /* The MSP stack is not reset as, unlike on M3/4 parts, there is no vector
167  table offset register that can be used to locate the initial stack value.
168  Not all M0 parts have the application vector table at address 0. */
169  __asm volatile(
170  " ldr r2, pxCurrentTCBConst2 \n" /* Obtain location of pxCurrentTCB. */
171  " ldr r3, [r2] \n"
172  " ldr r0, [r3] \n" /* The first item in pxCurrentTCB is the task top of stack. */
173  " add r0, #32 \n" /* Discard everything up to r0. */
174  " msr psp, r0 \n" /* This is now the new top of stack to use in the task. */
175  " movs r0, #2 \n" /* Switch to the psp stack. */
176  " msr CONTROL, r0 \n"
177  " isb \n"
178  " pop {r0-r5} \n" /* Pop the registers that are saved automatically. */
179  " mov lr, r5 \n" /* lr is now in r5. */
180  " pop {r3} \n" /* Return address is now in r3. */
181  " pop {r2} \n" /* Pop and discard XPSR. */
182  " cpsie i \n" /* The first task has its context and interrupts can be enabled. */
183  " bx r3 \n" /* Finally, jump to the user defined task code. */
184  " \n"
185  " .align 4 \n"
186  "pxCurrentTCBConst2: .word pxCurrentTCB "
187  );
188 }

194 BaseType_t xPortStartScheduler( void )
195 {
196  /* Make PendSV, CallSV and SysTick the same priroity as the kernel. */
197  *(portNVIC_SYSPRI2) |= portNVIC_PENDSV_PRI;
198  *(portNVIC_SYSPRI2) |= portNVIC_SYSTICK_PRI;
199 
200  /* Start the timer that generates the tick ISR. Interrupts are disabled
201  here already. */
202  prvSetupTimerInterrupt();
203 
204  /* Initialise the critical nesting count ready for the first task. */
205  uxCriticalNesting = 0;
206 
207  /* Start the first task. */
208  vPortStartFirstTask();
209 
210  /* Should never get here as the tasks will now be executing! Call the task
211  exit error function to prevent compiler warnings about a static function
212  not being called in the case that the application writer overrides this
213  functionality by defining configTASK_RETURN_ADDRESS. */
214  prvTaskExitError();
215 
216  /* Should not get here! */
217  return 0;
218 }

286 void xPortPendSVHandler( void )
287 {
288  /* This is a naked function. */
289 
290  __asm volatile
291  (
292  " mrs r0, psp \n"
293  " \n"
294  " ldr r3, pxCurrentTCBConst \n" /* Get the location of the current TCB. */
295  " ldr r2, [r3] \n"
296  " \n"
297  " sub r0, r0, #32 \n" /* Make space for the remaining low registers. */
298  " str r0, [r2] \n" /* Save the new top of stack. */
299  " stmia r0!, {r4-r7} \n" /* Store the low registers that are not saved automatically. */
300  " mov r4, r8 \n" /* Store the high registers. */
301  " mov r5, r9 \n"
302  " mov r6, r10 \n"
303  " mov r7, r11 \n"
304  " stmia r0!, {r4-r7} \n"
305  " \n"
306  " push {r3, r14} \n"
307  " cpsid i \n"
308  " bl vTaskSwitchContext \n"
309  " cpsie i \n"
310  " pop {r2, r3} \n" /* lr goes in r3. r2 now holds tcb pointer. */
311  " \n"
312  " ldr r1, [r2] \n"
313  " ldr r0, [r1] \n" /* The first item in pxCurrentTCB is the task top of stack. */
314  " add r0, r0, #16 \n" /* Move to the high registers. */
315  " ldmia r0!, {r4-r7} \n" /* Pop the high registers. */
316  " mov r8, r4 \n"
317  " mov r9, r5 \n"
318  " mov r10, r6 \n"
319  " mov r11, r7 \n"
320  " \n"
321  " msr psp, r0 \n" /* Remember the new top of stack for the task. */
322  " \n"
323  " sub r0, r0, #32 \n" /* Go back for the low registers that are not automatically restored. */
324  " ldmia r0!, {r4-r7} \n" /* Pop low registers. */
325  " \n"
326  " bx r3 \n"
327  " \n"
328  " .align 4 \n"
329  "pxCurrentTCBConst: .word pxCurrentTCB "
330  );
331 }
