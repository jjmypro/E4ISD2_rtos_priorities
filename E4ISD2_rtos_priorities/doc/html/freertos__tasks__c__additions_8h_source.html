58 /* NOTE!!
59  * The configFRTOS_MEMORY_SCHEME macro describes the heap scheme using a value
60  * 1 - 5 which corresponds to the following schemes:
61  *
62  * heap_1 - the very simplest, does not permit memory to be freed
63  * heap_2 - permits memory to be freed, but not does coalescence adjacent free
64  * blocks.
65  * heap_3 - simply wraps the standard malloc() and free() for thread safety
66  * heap_4 - coalesces adjacent free blocks to avoid fragmentation. Includes
67  * absolute address placement option
68  * heap_5 - as per heap_4, with the ability to span the heap across
69  * multiple nonOadjacent memory areas
70  */
71 #ifndef configFRTOS_MEMORY_SCHEME
72 #define configFRTOS_MEMORY_SCHEME 3 /* thread safe malloc */
73 #endif
74 
75 #if ((configFRTOS_MEMORY_SCHEME > 5) || (configFRTOS_MEMORY_SCHEME < 1))
76 #error "Invalid configFRTOS_MEMORY_SCHEME setting!"
