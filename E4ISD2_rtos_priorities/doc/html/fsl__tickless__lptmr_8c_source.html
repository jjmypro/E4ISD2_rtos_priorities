129  void vPortSuppressTicksAndSleep( TickType_t xExpectedIdleTime )
130  {
131  uint32_t ulReloadValue, ulCompleteTickPeriods;
132  TickType_t xModifiableIdleTime;
133  LPTMR_Type *pxLptmrBase;
134 
135  pxLptmrBase = vPortGetLptrmBase();
136  if (pxLptmrBase == 0) return;
137  /* Make sure the SysTick reload value does not overflow the counter. */
138  if( xExpectedIdleTime > xMaximumPossibleSuppressedTicks )
139  {
140  xExpectedIdleTime = xMaximumPossibleSuppressedTicks;
141  }
142  if (xExpectedIdleTime == 0) return;
143  /* Calculate the reload value required to wait xExpectedIdleTime
144  tick periods. -1 is used because this code will execute part way
145  through one of the tick periods. */
146  ulReloadValue = LPTMR_GetCurrentTimerCount(pxLptmrBase) + ( ulLPTimerCountsForOneTick * ( xExpectedIdleTime - 1UL ) );
147 
148  /* Stop the LPTMR and systick momentarily. The time the LPTMR and systick is stopped for
149  is accounted for as best it can be, but using the tickless mode will
150  inevitably result in some tiny drift of the time maintained by the
151  kernel with respect to calendar time. */
152  LPTMR_StopTimer(pxLptmrBase);
153  *(portNVIC_SYSTICK_CTRL) &= ~portNVIC_SYSTICK_ENABLE_BIT;
154 
155  /* Enter a critical section but don't use the taskENTER_CRITICAL()
156  method as that will mask interrupts that should exit sleep mode. */
157  __asm volatile( "cpsid i" );
158 
159  /* If a context switch is pending or a task is waiting for the scheduler
160  to be unsuspended then abandon the low power entry. */
161  if( eTaskConfirmSleepModeStatus() == eAbortSleep )
162  {
163  /* Restart from whatever is left in the count register to complete
164  this tick period. */
165  *(portNVIC_SYSTICK_LOAD) = portNVIC_SYSTICK_CURRENT_VALUE_REG;
166 
167  /* Restart SysTick. */
168  *(portNVIC_SYSTICK_CTRL) |= portNVIC_SYSTICK_ENABLE_BIT;
169 
170  /* Reset the reload register to the value required for normal tick
171  periods. */
172  *(portNVIC_SYSTICK_LOAD) = ulTimerCountsForOneTick - 1UL;
173 
174  /* Re-enable interrupts - see comments above __disable_interrupt()
175  call above. */
176  __asm volatile( "cpsie i" );
177  }
178  else
179  {
180  /* Set the new reload value. */
181  LPTMR_SetTimerPeriod(pxLptmrBase, ulReloadValue);
182 
183  /* Enable LPTMR. */
184  LPTMR_StartTimer(pxLptmrBase);
185 
186  /* Sleep until something happens. configPRE_SLEEP_PROCESSING() can
187  set its parameter to 0 to indicate that its implementation contains
188  its own wait for interrupt or wait for event instruction, and so wfi
189  should not be executed again. However, the original expected idle
190  time variable must remain unmodified, so a copy is taken. */
191  xModifiableIdleTime = xExpectedIdleTime;
192  configPRE_SLEEP_PROCESSING( xModifiableIdleTime );
193  if( xModifiableIdleTime > 0 )
194  {
195  __asm volatile( "dsb" );
196  __asm volatile( "wfi" );
197  __asm volatile( "isb" );
198  }
199  configPOST_SLEEP_PROCESSING( xExpectedIdleTime );
200 
201  ulLPTimerInterruptFired = false;
202  /* Re-enable interrupts - see comments above __disable_interrupt()
203  call above. */
204  __asm volatile( "cpsie i" );
205  __NOP();
206  if( ulLPTimerInterruptFired )
207  {
208 
209  /* The tick interrupt handler will already have pended the tick
210  processing in the kernel. As the pending tick will be
211  processed as soon as this function exits, the tick value
212  maintained by the tick is stepped forward by one less than the
213  time spent waiting. */
214  ulCompleteTickPeriods = xExpectedIdleTime - 1UL;
215  ulLPTimerInterruptFired = false;
216  }
217  else
218  {
219  /* Something other than the tick interrupt ended the sleep.
220  Work out how long the sleep lasted rounded to complete tick
221  periods (not the ulReload value which accounted for part
222  ticks). */
223  ulCompleteTickPeriods = LPTMR_GetCurrentTimerCount(pxLptmrBase)/ulLPTimerCountsForOneTick;
224 
225  }
226 
227  /* Stop LPTMR when CPU waked up then set portNVIC_SYSTICK_LOAD_REG back to its standard
228  value. The critical section is used to ensure the tick interrupt
229  can only execute once in the case that the reload register is near
230  zero. */
231  LPTMR_StopTimer(pxLptmrBase);
232  portENTER_CRITICAL();
233  {
234  *(portNVIC_SYSTICK_CTRL) |= portNVIC_SYSTICK_ENABLE_BIT;
235  vTaskStepTick( ulCompleteTickPeriods );
236  *(portNVIC_SYSTICK_LOAD) = ulTimerCountsForOneTick - 1UL;
237  }
238  portEXIT_CRITICAL();
239  }
240  }
241 #endif /* #if configUSE_TICKLESS_IDLE */

252  ulLPTimerCountsForOneTick = ( configLPTMR_CLOCK_HZ / configTICK_RATE_HZ );

255 #endif /* configUSE_TICKLESS_IDLE */
256  /* Configure SysTick to interrupt at the requested rate. */
257  *(portNVIC_SYSTICK_LOAD) = ( configCPU_CLOCK_HZ / configTICK_RATE_HZ ) - 1UL;
258  portNVIC_SYSTICK_CURRENT_VALUE_REG = 0UL;
